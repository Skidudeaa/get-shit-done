

Below is a complete, unified, global CLI design + implementation that gives you:
	•	codebase-intel retrieve "<query>" → stable JSON contract (Zoekt → fallback rg) + graph expansion + reranking
	•	codebase-intel hook sessionstart → prints <codebase-intelligence>…</codebase-intelligence> from the repo’s .planning/intel/summary.md
	•	codebase-intel watch / scan / init / health / summary
	•	Optional: codebase-intel zoekt index / zoekt serve / zoekt search

This is global tool code, per-repo state:
	•	Tool lives once (npm link / global install).
	•	Each repo has .planning/intel/* state (graph/index/summary) and optional .codebase-intel.json config.
	•	Each repo’s .claude/settings.json only needs to call the global binary.

const fs = require("fs");
const initSqlJs = require("sql.js/dist/sql-asm.js");
const { graphDbPath } = require("./config");

async function openGraph(root) {
  const p = graphDbPath(root);
  if (!fs.existsSync(p)) return null;

  const SQL = await initSqlJs();
  const buf = fs.readFileSync(p);
  const db = new SQL.Database(buf);
  return { db, path: p };
}

function closeGraph(g) {
  try { g?.db?.close(); } catch {}
}

function slugForPath(relPath) {
  return relPath
    .replace(/^\/+/, "")
    .replace(/\.[^.]+$/, "")
    .replace(/[\/\\]/g, "-")
    .replace(/[^a-zA-Z0-9-]/g, "-")
    .replace(/-+/g, "-")
    .replace(/^-|-$/g, "")
    .toLowerCase();
}

function getNodeMeta(db, ids) {
  if (!ids.length) return new Map();
  const placeholders = ids.map(() => "?").join(",");
  const r = db.exec(
    `SELECT id, body FROM nodes WHERE id IN (${placeholders})`,
    ids
  );

  const m = new Map();
  for (const row of (r?.[0]?.values || [])) {
    const [id, body] = row;
    try { m.set(id, JSON.parse(body)); } catch {}
  }
  return m;
}

function getFanIn(db, ids) {
  const out = new Map();
  if (!ids.length) return out;

  const placeholders = ids.map(() => "?").join(",");
  const r = db.exec(
    `SELECT target, COUNT(*) FROM edges WHERE target IN (${placeholders}) GROUP BY target`,
    ids
  );
  for (const [id, c] of (r?.[0]?.values || [])) out.set(id, c);
  return out;
}

function getFanOut(db, ids) {
  const out = new Map();
  if (!ids.length) return out;

  const placeholders = ids.map(() => "?").join(",");
  const r = db.exec(
    `SELECT source, COUNT(*) FROM edges WHERE source IN (${placeholders}) GROUP BY source`,
    ids
  );
  for (const [id, c] of (r?.[0]?.values || [])) out.set(id, c);
  return out;
}

function neighbors(db, id, { maxImports = 20, maxDependents = 20 } = {}) {
  const out = db.exec(`SELECT target FROM edges WHERE source = ? LIMIT ?`, [id, maxImports]);
  const inn = db.exec(`SELECT source FROM edges WHERE target = ? LIMIT ?`, [id, maxDependents]);

  return {
    imports: (out?.[0]?.values || []).map(v => v[0]),
    dependents: (inn?.[0]?.values || []).map(v => v[0])
  };
}

src/rg.js (fallback search backend)

const { spawnSync, spawn } = require("child_process");

function which(bin) {
  const r = spawnSync("sh", ["-lc", `command -v ${bin} 2>/dev/null`], { encoding: "utf8" });
  if (r.status === 0) {
    const out = (r.stdout || "").trim();
    return out ? out : null;
  }
  return null;
}

function isInstalled() {
  return !!which("rg");
}

function readFileLines(path) {
  const fs = require("fs");
  try {
    const txt = fs.readFileSync(path, "utf8");
    return txt.split(/\r?\n/);
  } catch {
    return null;
  }
}

function addContextToHits(root, hits, contextLines) {
  if (!contextLines || contextLines <= 0) {
    for (const h of hits) { h.before = []; h.after = []; }
    return hits;
  }

  const path = require("path");
  const byFile = new Map();
  for (const h of hits) {
    if (!byFile.has(h.path)) byFile.set(h.path, []);
    byFile.get(h.path).push(h);
  }

  for (const [rel, fileHits] of byFile.entries()) {
    const abs = path.join(root, rel);
    const lines = readFileLines(abs);
    if (!lines) {
      for (const h of fileHits) { h.before = []; h.after = []; }
      continue;
    }

    for (const h of fileHits) {
      const idx = (h.lineNumber || 1) - 1;
      const start = Math.max(0, idx - contextLines);
      const end = Math.min(lines.length - 1, idx + contextLines);

      h.before = lines.slice(start, idx);
      h.after = lines.slice(idx + 1, end + 1);
    }
  }

  return hits;
}

async function search(root, q, opts = {}) {
  if (!isInstalled()) throw new Error("rg not found in PATH");

  // default to literal search to avoid regex surprises
  const mode = opts.mode || "literal"; // "literal" | "regex"
  const maxHits = opts.maxHits ?? 50;
  const contextLines = opts.contextLines ?? 1;

  const args = [
    "--json",
    "-n",
    "--smart-case"
  ];

  // keep our own state out of results
  args.push("--glob", "!.planning/**");

  if (mode === "literal") args.push("-F");

  args.push(q);
  args.push("."); // search from root

  const hits = [];

  await new Promise((resolve, reject) => {
    const child = spawn("rg", args, { cwd: root });

    let buf = "";
    child.stdout.setEncoding("utf8");
    child.stdout.on("data", (chunk) => {
      buf += chunk;
      let nl;
      while ((nl = buf.indexOf("\n")) >= 0) {
        const line = buf.slice(0, nl);
        buf = buf.slice(nl + 1);

        if (!line.trim()) continue;
        let evt;
        try { evt = JSON.parse(line); } catch { continue; }

        if (evt.type === "match") {
          const data = evt.data;
          const pathText = data.path?.text;
          const ln = data.line_number;
          const text = data.lines?.text?.replace(/\n$/, "") ?? "";
          const subs = (data.submatches || []).map(sm => ({
            start: sm.start,
            end: sm.end
          }));

          hits.push({
            path: pathText,
            lineNumber: ln,
            line: text,
            ranges: subs,
            provider: "rg",
            score: null
          });

          if (hits.length >= maxHits) {
            child.kill("SIGTERM");
            break;
          }
        }
      }
    });

    child.on("error", reject);
    child.on("exit", () => resolve());
  });

  addContextToHits(root, hits, contextLines);

  return {
    provider: "rg",
    stats: { matchCount: hits.length },
    hits
  };
}

module.exports = { isInstalled, search };

src/zoekt.js (search backend + local daemon)

const fs = require("fs");
const path = require("path");
const { spawn, spawnSync } = require("child_process");
const { zoektIndexDir, zoektDaemonPath, zoektDir } = require("./config");

function exists(p) {
  try { fs.accessSync(p); return true; } catch { return false; }
}
function mkdirp(p) { fs.mkdirSync(p, { recursive: true }); }

function which(bin) {
  const r = spawnSync("sh", ["-lc", `command -v ${bin} 2>/dev/null`], { encoding: "utf8" });
  if (r.status === 0) {
    const out = (r.stdout || "").trim();
    return out ? out : null;
  }
  return null;
}

function isInstalled() {
  // minimal: webserver must exist for API mode
  return !!which("zoekt-webserver");
}

function isGitRepo(root) {
  return exists(path.join(root, ".git"));
}

function hasIndex(root) {
  const dir = zoektIndexDir(root);
  if (!exists(dir)) return false;
  return fs.readdirSync(dir).some(f => f.endsWith(".zoekt"));
}

function readDaemon(root) {
  const p = zoektDaemonPath(root);
  if (!exists(p)) return null;
  try { return JSON.parse(fs.readFileSync(p, "utf8")); } catch { return null; }
}

function writeDaemon(root, data) {
  mkdirp(zoektDir(root));
  fs.writeFileSync(zoektDaemonPath(root), JSON.stringify(data, null, 2));
}

function isPidRunning(pid) {
  if (!pid || typeof pid !== "number") return false;
  try { process.kill(pid, 0); return true; } catch { return false; }
}

function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }

async function probeApi(port) {
  const url = `http://127.0.0.1:${port}/api/search`;
  try {
    const res = await fetch(url, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ Q: "file:.", Opts: { TotalMaxMatchCount: 1, NumContextLines: 0 } })
    });
    return res.ok;
  } catch {
    return false;
  }
}

function buildIndex(root, { force = false } = {}) {
  mkdirp(zoektIndexDir(root));
  if (!force && hasIndex(root)) return { ok: true, skipped: true };

  const bin = isGitRepo(root) ? "zoekt-git-index" : "zoekt-index";
  if (!which(bin)) throw new Error(`Missing ${bin} in PATH`);

  const args = ["-index", zoektIndexDir(root), root];
  const r = spawnSync(bin, args, { encoding: "utf8" });
  if (r.status !== 0) {
    throw new Error(`Zoekt index failed: ${bin} ${args.join(" ")}\n${r.stderr || r.stdout || ""}`);
  }
  return { ok: true, indexed: true };
}

async function ensureWebserver(root, { port = 6070, autoIndex = false } = {}) {
  if (!isInstalled()) throw new Error("zoekt-webserver not found in PATH");

  const df = readDaemon(root);
  if (df?.pid && df?.port && isPidRunning(df.pid)) {
    if (await probeApi(df.port)) return { ok: true, reused: true, port: df.port, pid: df.pid };
  }

  if (!hasIndex(root)) {
    if (!autoIndex) throw new Error("Zoekt index missing; run `codebase-intel zoekt index` or pass --zoekt-build");
    buildIndex(root, { force: false });
  }

  // try ports
  const ports = [port, ...Array.from({ length: 30 }, (_, i) => port + i + 1)];
  mkdirp(zoektDir(root));
  const lf = path.join(zoektDir(root), "webserver.log");
  const out = fs.openSync(lf, "a");

  for (const p of ports) {
    // best-effort: use -listen if supported
    const args = ["-rpc", "-index", zoektIndexDir(root), "-listen", `127.0.0.1:${p}`];
    const child = spawn("zoekt-webserver", args, {
      cwd: root,
      detached: true,
      stdio: ["ignore", out, out]
    });
    child.unref();

    await sleep(250);
    if (await probeApi(p)) {
      writeDaemon(root, { pid: child.pid, port: p, startedAt: new Date().toISOString() });
      return { ok: true, started: true, port: p, pid: child.pid };
    }

    try { process.kill(child.pid, "SIGTERM"); } catch {}
    await sleep(75);

    // fallback once at default port with no -listen
    if (p === 6070) {
      const child2 = spawn("zoekt-webserver", ["-rpc", "-index", zoektIndexDir(root)], {
        cwd: root,
        detached: true,
        stdio: ["ignore", out, out]
      });
      child2.unref();

      await sleep(300);
      if (await probeApi(6070)) {
        writeDaemon(root, { pid: child2.pid, port: 6070, startedAt: new Date().toISOString() });
        return { ok: true, started: true, port: 6070, pid: child2.pid };
      }

      try { process.kill(child2.pid, "SIGTERM"); } catch {}
      await sleep(75);
    }
  }

  throw new Error("Failed to start zoekt-webserver (rpc). Check .planning/intel/zoekt/webserver.log");
}

function b64(s) {
  if (!s) return "";
  try { return Buffer.from(s, "base64").toString("utf8"); } catch { return s; }
}

function normalize(json) {
  const result = json?.Result || json?.result || json;
  const files = result?.Files || result?.files || [];

  const hits = [];
  for (const f of files) {
    const fileName = f.FileName || f.fileName;
    const fileScore = f.Score || f.score || null;
    const lineMatches = f.LineMatches || f.lineMatches || [];

    for (const lm of lineMatches) {
      const ranges = (lm.LineFragments || lm.lineFragments || [])
        .filter(fr => fr.Match)
        .map(fr => ({ start: fr.Start, end: fr.End }));

      hits.push({
        path: fileName,
        lineNumber: lm.LineNumber ?? lm.lineNumber ?? null,
        line: b64(lm.Line).replace(/\n$/, ""),
        before: b64(lm.Before).split(/\r?\n/).filter(Boolean),
        after: b64(lm.After).split(/\r?\n/).filter(Boolean),
        ranges,
        provider: "zoekt",
        score: lm.Score ?? lm.score ?? fileScore
      });
    }
  }

  return {
    provider: "zoekt",
    stats: {
      matchCount: hits.length,
      fileCount: result?.FileCount ?? result?.fileCount ?? files.length ?? null
    },
    hits
  };
}

async function search(root, q, opts = {}) {
  const { port } = await ensureWebserver(root, {
    port: opts.port ?? 6070,
    autoIndex: !!opts.autoIndex
  });

  const url = `http://127.0.0.1:${port}/api/search`;
  const payload = {
    Q: q,
    Opts: {
      TotalMaxMatchCount: opts.totalMaxMatchCount ?? 500,
      NumContextLines: opts.contextLines ?? 1
    }
  };

  const res = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(payload)
  });

  if (!res.ok) throw new Error(`Zoekt search failed: HTTP ${res.status}`);
  const json = await res.json();
  const norm = normalize(json);

  // enforce maxHits client-side
  const maxHits = opts.maxHits ?? 50;
  if (norm.hits.length > maxHits) norm.hits = norm.hits.slice(0, maxHits);

  return { ...norm, details: { port } };
}

function extractImports(content) {
  const imports = new Set();

  const es6Named = /import\s+(?:\{[^}]*\}|\*\s+as\s+\w+|\w+)\s+from\s+['"]([^'"]+)['"]/g;
  const es6SideEffect = /import\s+['"]([^'"]+)['"]/g;
  const cjs = /require\s*\(\s*['"]([^'"]+)['"]\s*\)/g;
  const dyn = /\bimport\s*\(\s*['"]([^'"]+)['"]\s*\)/g;
  const reExportAll = /export\s+\*\s+from\s+['"]([^'"]+)['"]/g;
  const reExportNamed = /export\s+\{[^}]*\}\s+from\s+['"]([^'"]+)['"]/g;

  let m;
  while ((m = es6Named.exec(content)) !== null) imports.add(m[1]);

  while ((m = es6SideEffect.exec(content)) !== null) {
    if (!content.slice(Math.max(0, m.index - 10), m.index).includes("from")) {
      imports.add(m[1]);
    }
  }

  while ((m = cjs.exec(content)) !== null) imports.add(m[1]);
  while ((m = dyn.exec(content)) !== null) imports.add(m[1]);
  while ((m = reExportAll.exec(content)) !== null) imports.add(m[1]);
  while ((m = reExportNamed.exec(content)) !== null) imports.add(m[1]);

  return [...imports];
}

function extractExports(content) {
  const exports = new Set();
  let match;

  const namedExport = /export\s*\{([^}]+)\}/g;
  while ((match = namedExport.exec(content)) !== null) {
    match[1].split(",").map(n => {
      const parts = n.trim().split(/\s+as\s+/);
      return parts[parts.length - 1].trim();
    }).filter(Boolean).forEach(n => exports.add(n));
  }

  const declExport = /export\s+(?:const|let|var|function\*?|async\s+function|class)\s+(\w+)/g;
  while ((match = declExport.exec(content)) !== null) exports.add(match[1]);

  const defaultExport = /export\s+default\s+(?:function\s*\*?\s*|class\s+)?(\w+)?/g;
  while ((match = defaultExport.exec(content)) !== null) {
    exports.add("default");
    if (match[1]) exports.add(match[1]);
  }

  const cjsExport = /module\.exports\s*=\s*\{([^}]+)\}/g;
  while ((match = cjsExport.exec(content)) !== null) {
    match[1].split(",").map(n => n.trim().split(/\s*:\s*/)[0].trim())
      .filter(n => n && /^\w+$/.test(n)).forEach(n => exports.add(n));
  }

  const cjsSingleExport = /module\.exports\s*=\s*(\w+)\s*[;\n]/g;
  while ((match = cjsSingleExport.exec(content)) !== null) {
    exports.add("default");
    exports.add(match[1]);
  }

  const tsExport = /export\s+(?:type|interface)\s+(\w+)/g;
  while ((match = tsExport.exec(content)) !== null) exports.add(match[1]);

  return [...exports];
}

// ---- tsconfig paths ----
function loadTSConfig(root) {
  const st = S(root);
  if (st.tsconfig) return st.tsconfig;

  const configPath = path.join(root, "tsconfig.json");
  if (!exists(configPath)) return null;

  let raw;
  try { raw = JSON.parse(fs.readFileSync(configPath, "utf8")); } catch { return null; }

  const compilerOptions = raw.compilerOptions || {};
  const baseUrl = compilerOptions.baseUrl ? path.resolve(root, compilerOptions.baseUrl) : root;
  const paths = compilerOptions.paths || {};

  const pathMatchers = Object.entries(paths).map(([alias, targets]) => {
    const regex = new RegExp("^" + alias
      .replace(/[.+?^${}()|[\]\\]/g, "\\$&")
      .replace("\\*", "(.+)") + "$");

    return {
      alias,
      regex,
      targets: (targets || []).map(t => path.resolve(baseUrl, t.replace("*", "$1")))
    };
  });

  st.tsconfig = { baseUrl, pathMatchers };
  return st.tsconfig;
}

function matchesTSPathAlias(root, spec) {
  const cfg = loadTSConfig(root);
  if (!cfg?.pathMatchers) return false;
  return cfg.pathMatchers.some(m => m.regex.test(spec));
}

function resolveTSPathAlias(root, spec) {
  const cfg = loadTSConfig(root);
  if (!cfg) return null;

  for (const { regex, targets } of cfg.pathMatchers) {
    const m = spec.match(regex);
    if (!m) continue;

    for (const target of targets) {
      const substituted = target.replace("$1", m[1] || "");
      const cands = [];

      if (path.extname(substituted)) cands.push(substituted);
      else {
        for (const ext of INDEXABLE_EXTS) cands.push(substituted + ext);
        for (const ext of INDEXABLE_EXTS) cands.push(path.join(substituted, "index" + ext));
      }

      for (const c of cands) if (exists(c)) return c;
    }
  }

  return null;
}

// ---- workspace packages ----
function loadWorkspacePackages(root) {
  const st = S(root);
  if (st.workspaces) return st.workspaces;

  const pkgPath = path.join(root, "package.json");
  if (!exists(pkgPath)) return (st.workspaces = []);

  let pkg;
  try { pkg = JSON.parse(fs.readFileSync(pkgPath, "utf8")); } catch { return (st.workspaces = []); }

  const workspaces = pkg.workspaces?.packages || pkg.workspaces || [];
  const globs = Array.isArray(workspaces) ? workspaces : [];

  const dirs = globs.flatMap(g => fg.sync(g, { cwd: root, onlyDirectories: true }));
  const pkgs = [];

  for (const d of dirs) {
    const p = path.join(root, d, "package.json");
    if (!exists(p)) continue;

    try {
      const pj = JSON.parse(fs.readFileSync(p, "utf8"));
      if (!pj.name) continue;
      pkgs.push({
        name: pj.name,
        dir: path.join(root, d),
        main: pj.source || pj.module || pj.main || null
      });
    } catch {}
  }

  st.workspaces = pkgs;
  return pkgs;
}

function matchesWorkspaceImport(root, spec) {
  const pkgs = loadWorkspacePackages(root);
  if (!pkgs.length) return false;

  if (spec.startsWith("@")) {
    const parts = spec.split("/");
    if (parts.length < 2) return false;
    const name = `${parts[0]}/${parts[1]}`;
    return pkgs.some(p => p.name === name) || pkgs.some(p => spec.startsWith(name + "/"));
  }

  const name = spec.split("/")[0];
  return pkgs.some(p => p.name === name) || pkgs.some(p => spec.startsWith(name + "/"));
}

function resolveWorkspaceImport(root, spec) {
  const pkgs = loadWorkspacePackages(root);
  if (!pkgs.length) return null;

  let pkgName = null;
  let subpath = "";

  if (spec.startsWith("@")) {
    const parts = spec.split("/");
    if (parts.length < 2) return null;
    pkgName = `${parts[0]}/${parts[1]}`;
    subpath = parts.slice(2).join("/");
  } else {
    pkgName = spec.split("/")[0];
    subpath = spec.slice(pkgName.length + 1);
    if (subpath === spec) subpath = ""; // no slash case
  }

  const pkg = pkgs.find(p => p.name === pkgName);
  if (!pkg) return null;

  let base;
  if (!subpath) {
    // try declared entry; else common source defaults
    base = pkg.main
      ? path.join(pkg.dir, pkg.main)
      : path.join(pkg.dir, "src", "index");
  } else {
    base = path.join(pkg.dir, subpath);
  }

  const cands = [];
  if (path.extname(base)) cands.push(base);
  else {
    for (const ext of INDEXABLE_EXTS) cands.push(base + ext);
    for (const ext of INDEXABLE_EXTS) cands.push(path.join(base, "index" + ext));
  }

  for (const c of cands) if (exists(c)) return c;
  return null;
}

function isLocalishImport(root, spec) {
  return spec.startsWith(".") || spec.startsWith("/") || matchesTSPathAlias(root, spec) || matchesWorkspaceImport(root, spec);
}

function resolveLocalImport(fromFileAbs, spec, root) {
  // 1) relative/absolute
  if (spec.startsWith(".") || spec.startsWith("/")) {
    const fromDir = path.dirname(fromFileAbs);
    const base = spec.startsWith("/")
      ? path.resolve(root, spec.slice(1))
      : path.resolve(fromDir, spec);

    const cands = [];
    if (path.extname(base)) cands.push(base);
    else {
      for (const ext of INDEXABLE_EXTS) cands.push(base + ext);
      for (const ext of INDEXABLE_EXTS) cands.push(path.join(base, "index" + ext));
    }

    for (const c of cands) if (exists(c)) return c;
    return null;
  }

  // 2) tsconfig paths
  const aliased = resolveTSPathAlias(root, spec);
  if (aliased) return aliased;

  // 3) workspace packages
  const ws = resolveWorkspaceImport(root, spec);
  if (ws) return ws;

  return null;
}

function fileTypeHeuristic(relPath) {
  const p = relPath.toLowerCase();

  if (p.includes("/components/") || p.endsWith(".tsx") || p.endsWith(".jsx")) return "component";
  if (p.includes("/hooks/") || p.includes(".hook.")) return "hook";
  if (p.includes("/utils/") || p.includes("/lib/")) return "util";
  if (p.includes("/services/")) return "service";
  if (p.includes("/api/") || p.includes("/routes/")) return "api";
  if (p.includes("/models/") || p.includes("/types/")) return "model";
  if (p.includes("/config/")) return "config";
  return "module";
}

async function updateFile(root, relPath, opts = {}) {
  const abs = path.resolve(root, relPath);
  if (!exists(abs)) return { skipped: true, reason: "missing" };
  if (!isCodeFile(abs)) return { skipped: true, reason: "not-code" };

  const content = fs.readFileSync(abs, "utf8");
  const imports = extractImports(content);
  const exports = extractExports(content);

  // resolution health + resolved targets
  let localTotal = 0;
  let localResolved = 0;
  const unresolved = [];
  const resolvedTargets = [];

  for (const spec of imports) {
    if (!isLocalishImport(root, spec)) continue;
    localTotal++;

    const targetAbs = resolveLocalImport(abs, spec, root);
    if (!targetAbs) {
      unresolved.push(spec);
      continue;
    }

    localResolved++;
    const targetRel = path.relative(root, targetAbs);
    resolvedTargets.push(targetRel);
  }

  // index.json update
  let index = { version: 1, updated: null, files: {} };
  try { index = JSON.parse(fs.readFileSync(indexPath(root), "utf8")); } catch {}

  index.files[abs] = {
    exports,
    imports,
    indexed: Date.now(),
    metrics: {
      local_imports_total: localTotal,
      local_imports_resolved: localResolved,
      local_imports_unresolved: unresolved
    }
  };
  index.updated = Date.now();
  fs.writeFileSync(indexPath(root), JSON.stringify(index, null, 2));

  // graph update
  const id = slugForPath(relPath);
  const { db, path: p } = await loadDb(root);

  const nodeBody = JSON.stringify({
    id,
    path: relPath,
    type: fileTypeHeuristic(relPath),
    updated: new Date().toISOString().split("T")[0],
    status: "active",
    metrics: {
      exports_total: exports.length,
      imports_total: imports.length,
      local_imports_total: localTotal,
      local_imports_resolved: localResolved
    }
  });

  db.run(
    `INSERT INTO nodes (body) VALUES (?) ON CONFLICT(id) DO UPDATE SET body = excluded.body`,
    [nodeBody]
  );

  // replace edges for this file
  db.run(`DELETE FROM edges WHERE source = ?`, [id]);

  if (resolvedTargets.length) {
    const stmt = db.prepare(`INSERT INTO edges (source, target, relationship) VALUES (?, ?, 'imports')`);
    for (const targetRel of resolvedTargets) {
      const tid = slugForPath(targetRel);
      stmt.run([id, tid]);
    }
    stmt.free();
  }

  persistDb(db, p);
  db.close();

  await maybeWriteSummary(root, opts);

  return {
    indexed: true,
    path: relPath,
    id,
    imports: imports.length,
    exports: exports.length,
    edges: resolvedTargets.length,
    health: localTotal ? Math.round((localResolved * 100) / localTotal) : 100
  };
}

async function scan(root, globPattern, opts = {}) {
  await init(root);

  const files = await fg(globPattern, {
    cwd: root,
    dot: false,
    ignore: opts.ignore || [],
    onlyFiles: true
  });

  let indexed = 0, skipped = 0, errors = 0;
  for (const rel of files) {
    try {
      const r = await updateFile(root, rel, { skipSummary: true });
      if (r.indexed) indexed++;
      else skipped++;
    } catch {
      errors++;
    }
  }

  // always write summary at end
  await writeSummary(root, { force: true });

  return { indexed, skipped, errors, files: files.length };
}

function unresolvedKey(spec) {
  if (spec.startsWith("@/")) return "@/";
  if (spec.startsWith("~/")) return "~/";
  if (spec.startsWith("@")) {
    const parts = spec.split("/");
    if (parts.length >= 2) return `${parts[0]}/${parts[1]}`;
    return parts[0];
  }
  return spec.split("/")[0];
}

function health(root) {
  let index = { files: {}, updated: null };
  try { index = JSON.parse(fs.readFileSync(indexPath(root), "utf8")); } catch {}

  let total = 0;
  let resolved = 0;
  const miss = new Map();

  for (const entry of Object.values(index.files || {})) {
    const m = entry.metrics;
    if (!m) continue;
    total += m.local_imports_total || 0;
    resolved += m.local_imports_resolved || 0;
    for (const spec of (m.local_imports_unresolved || [])) {
      const k = unresolvedKey(spec);
      miss.set(k, (miss.get(k) || 0) + 1);
    }
  }

  const pct = total > 0 ? Math.round((resolved * 100) / total) : 100;
  const updatedMs = index.updated || null;
  const ageSec = updatedMs ? Math.floor((Date.now() - updatedMs) / 1000) : null;
  const topMisses = [...miss.entries()].sort((a,b)=>b[1]-a[1]).slice(0, 5);

  return {
    resolutionPct: pct,
    localResolved: resolved,
    localTotal: total,
    indexAgeSec: ageSec,
    topMisses
  };
}

function clampChars(s, max = 2200) {
  if (s.length <= max) return s;
  return s.slice(0, max - 20).replace(/\s+\S*$/, "") + "\n…(truncated)";
}

function getGitInfo(root) {
  if (!exists(path.join(root, ".git"))) return null;
  try {
    const branch = execSync(`git -C "${root.replace(/"/g, '\\"')}" rev-parse --abbrev-ref HEAD`, { encoding: "utf8" }).trim();
    const head = execSync(`git -C "${root.replace(/"/g, '\\"')}" rev-parse HEAD`, { encoding: "utf8" }).trim();
    return { branch, head };
  } catch {
    return null;
  }
}

function getRecentGitFiles(root, limit = 5) {
  if (!exists(path.join(root, ".git"))) return [];
  try {
    const out = execSync(
      `git -C "${root.replace(/"/g, '\\"')}" log -n ${limit} --pretty=format:%ct --name-only --diff-filter=AMR`,
      { encoding: "utf8", stdio: ["ignore", "pipe", "ignore"] }
    );

    const lines = out.split("\n").map(s => s.trim()).filter(Boolean);
    const changes = [];
    let ts = null;

    for (const line of lines) {
      if (/^\d{9,}$/.test(line)) { ts = parseInt(line, 10); continue; }
      if (!ts) continue;
      changes.push({ file: line, ts });
    }

    // newest per file
    const seen = new Set();
    const uniq = [];
    for (const c of changes.sort((a,b)=>b.ts-a.ts)) {
      if (seen.has(c.file)) continue;
      seen.add(c.file);
      uniq.push(c);
      if (uniq.length >= limit) break;
    }
    return uniq;
  } catch {
    return [];
  }
}

async function writeSummary(root, opts = {}) {
  const h = health(root);

  const lines = [];
  const nowIso = new Date().toISOString();

  // loud banners
  if (h.localTotal > 0 && h.resolutionPct < 90) {
    lines.push(`ALERT: CODEBASE INTEL HEALTH FAIL — local import resolution ${h.resolutionPct}% (${h.localResolved}/${h.localTotal})`);
    lines.push("");
  }
  if (h.indexAgeSec != null && h.indexAgeSec > 24 * 3600) {
    lines.push(`ALERT: INDEX STALE — last update ${(h.indexAgeSec/3600).toFixed(1)}h ago (watcher dead?)`);
    lines.push("");
  }

  lines.push("# Codebase Intelligence");
  lines.push(`Updated: ${nowIso}`);

  const git = getGitInfo(root);
  if (git) lines.push(`Git: ${git.branch} @ ${git.head.slice(0, 12)}`);

  if (h.localTotal > 0) {
    const misses = h.topMisses.map(([k,c]) => `${k}×${c}`).join(", ");
    lines.push(`Health: local import resolution ${h.resolutionPct}% (${h.localResolved}/${h.localTotal}), index age ${h.indexAgeSec ?? "?"}s`);
    if (misses) lines.push(`Misses: ${misses}`);
  }
  lines.push("");

  // Pull a few structural signals from graph if present
  const { db } = await loadDb(root);
  try {
    const typeRes = db.exec(`
      SELECT json_extract(body,'$.type') as type, COUNT(*) as count
      FROM nodes GROUP BY type ORDER BY count DESC
    `);

    if (typeRes?.[0]?.values?.length) {
      lines.push("## Module Types");
      for (const [t,c] of typeRes[0].values.slice(0, 6)) lines.push(`- ${t || "unknown"}: ${c}`);
      lines.push("");
    }

    // hotspots (fan-in)
    const hotRes = db.exec(`
      SELECT e.target, COUNT(*) as c, json_extract(n.body,'$.path') as path
      FROM edges e LEFT JOIN nodes n ON e.target = n.id
      GROUP BY e.target ORDER BY c DESC LIMIT 5
    `);
    if (hotRes?.[0]?.values?.length) {
      lines.push("## Dependency Hotspots");
      for (const [id,c,p] of hotRes[0].values) lines.push(`- \`${p || id}\` (fan-in ${c})`);
      lines.push("");
    }

    // entrypoints (heuristic)
    const entryRes = db.exec(`SELECT json_extract(body,'$.path') as path, json_extract(body,'$.type') as type FROM nodes`);
    const entries = [];
    for (const [p,t] of (entryRes?.[0]?.values || [])) {
      if (p && isEntryPoint(p)) entries.push({ path: p, type: t || "unknown" });
    }
    if (entries.length) {
      lines.push("## Likely Entry Points");
      for (const e of entries.slice(0, 5)) lines.push(`- \`${e.path}\` (${e.type})`);
      lines.push("");
    }
  } finally {
    db.close();
  }

  const recent = getRecentGitFiles(root, 5);
  if (recent.length) {
    lines.push("## Recent Changes (git)");
    for (const r of recent) {
      const d = new Date(r.ts * 1000).toISOString().split("T")[0];
      lines.push(`- ${d} \`${r.file}\``);
    }
    lines.push("");
  }

  const out = clampChars(lines.join("\n"), 2200);
  fs.writeFileSync(summaryPath(root), out);
  return out;
}

async function maybeWriteSummary(root, opts = {}) {
  if (opts.skipSummary) return;

  const st = S(root);
  const now = Date.now();
  const throttleMs = (opts.summaryThrottleMs ?? (parseInt(process.env.INTEL_SUMMARY_THROTTLE_MS || "0", 10) || 0)) | 0;

  if (opts.force || throttleMs <= 0 || (now - st.lastSummaryTimeMs) >= throttleMs) {
    await writeSummary(root);
    st.lastSummaryTimeMs = now;
  }
}

function readSummary(root) {
  const p = summaryPath(root);
  if (!exists(p)) return null;
  try { return fs.readFileSync(p, "utf8"); } catch { return null; }
}

  if (!p) return false;
  return (
    /(^|\/)(__tests__|__test__|tests?|specs?)\//i.test(p) ||
    /\.(test|spec)\./i.test(p)
  );
}

function isVendorPath(p) {
  if (!p) return false;
  return /(^|\/)(node_modules|dist|build|coverage|\.next|out|vendor)\//i.test(p);
}

function looksLikeExportLine(line) {
  return typeof line === "string" && /^\s*export\b/.test(line);
}

function looksLikeDefinitionLine(line) {
  if (typeof line !== "string") return false;
  return (
    /^\s*(export\s+)?(default\s+)?(async\s+)?function\b/.test(line) ||
    /^\s*(export\s+)?(default\s+)?class\b/.test(line) ||
    /^\s*export\s+(const|let|var|type|interface)\b/.test(line)
  );
}

function safeNum(n, fallback = 0) {
  return Number.isFinite(n) ? n : fallback;
}

function uniq(arr) {
  return Array.from(new Set(arr));
}

function groupHitsByFile(hits) {
  const m = new Map();
  for (const h of hits) {
    if (!h.path) continue;
    if (!m.has(h.path)) m.set(h.path, []);
    m.get(h.path).push(h);
  }
  return m;
}

function pickBackend(backendOpt, root, { zoektBuild = false } = {}) {
  if (backendOpt === "zoekt") return "zoekt";
  if (backendOpt === "rg") return "rg";

  // auto
  if (zoekt.isInstalled()) {
    if (zoekt.hasIndex(root) || zoektBuild) return "zoekt";
  }
  return "rg";
}

/**
 * File-level rerank (seed file selection + display ordering)
 */
function rerankFiles(files, { useGraphBoost = true } = {}) {
  return files.sort((a, b) => {
    const ea = a.isEntryPoint ? 1 : 0;
    const eb = b.isEntryPoint ? 1 : 0;
    if (ea !== eb) return eb - ea;

    if (useGraphBoost) {
      const fa = safeNum(a.fanIn, 0);
      const fb = safeNum(b.fanIn, 0);
      if (fa !== fb) return fb - fa;
    }

    const ha = safeNum(a.hitCount, 0);
    const hb = safeNum(b.hitCount, 0);
    if (ha !== hb) return hb - ha;

    // Prefer bestAdjustedHitScore when available; otherwise fall back to provider score.
    const sa = safeNum(a.bestAdjustedHitScore, safeNum(a.bestHitScore, -Infinity));
    const sb = safeNum(b.bestAdjustedHitScore, safeNum(b.bestHitScore, -Infinity));
    return sb - sa;
  });
}

/**
 * Compute a hit score that respects provider scoring but nudges toward:
 *  - entrypoints
 *  - dependency hotspots (fan-in)
 *  - non-test, non-generated code
 *  - definition/export lines
 *
 * Uses small percent boosts so it scales with Zoekt score.
 */
function computeAdjustedHitScore(hit, fileMeta, opts) {
  const explain = !!opts.explainHits;
  const useGraphBoost = opts.useGraphBoost !== false;

  // Base score:
  // - Zoekt usually provides numeric score.
  // - rg has no score -> treat as 1 so multipliers behave.
  const base = Number.isFinite(hit.score) ? hit.score : 1;
  let adjusted = base;

  const signals = [];

  if (fileMeta) {
    // entrypoint boost (non-graph)
    if (fileMeta.isEntryPoint) {
      adjusted += base * 0.10;
      if (explain) signals.push("entrypoint:+10%");
    }

    // graph-dependent boosts
    if (useGraphBoost) {
      if (fileMeta.isHotspot) {
        adjusted += base * 0.15;
        if (explain) signals.push("hotspot:+15%");
      }

      const fanIn = safeNum(fileMeta.fanIn, 0);
      if (fanIn > 0) {
        const frac = Math.min(0.15, Math.log1p(fanIn) * 0.02); // up to +15%
        adjusted += base * frac;
        if (explain) signals.push(`fanin:${fanIn}:+${Math.round(frac * 100)}%`);
      }
    }

    // penalties (non-graph)
    if (isTestPath(fileMeta.path)) {
      adjusted -= base * 0.25;
      if (explain) signals.push("test:-25%");
    }

    if (isVendorPath(fileMeta.path)) {
      adjusted -= base * 0.50;
      if (explain) signals.push("vendor:-50%");
    }
  }

  // Line-content nudges
  if (looksLikeExportLine(hit.line)) {
    adjusted += base * 0.05;
    if (explain) signals.push("exportline:+5%");
  }
  if (looksLikeDefinitionLine(hit.line)) {
    adjusted += base * 0.03;
    if (explain) signals.push("defline:+3%");
  }

  // Small additive nudges so rg results don’t all tie
  if (!Number.isFinite(hit.score)) {
    if (fileMeta?.isEntryPoint) adjusted += 0.20;
    if (useGraphBoost && fileMeta?.isHotspot) adjusted += 0.25;
    if (useGraphBoost) adjusted += Math.min(0.25, Math.log1p(safeNum(fileMeta?.fanIn, 0)) * 0.05);
  }

  adjusted = Math.max(0, adjusted);
  return explain ? { adjustedScore: adjusted, signals } : { adjustedScore: adjusted };
}

/**
 * Sort hits by adjustedScore and cap hits per file for diversity.
 */
function rerankAndCapHits(hits, fileByPath, opts) {
  const maxHits = safeNum(opts.maxHits, hits.length);
  const cap = Number.isFinite(opts.hitsPerFileCap) ? opts.hitsPerFileCap : 5;

  const scored = hits.map((h) => {
    const fm = fileByPath.get(h.path);
    const extra = computeAdjustedHitScore(h, fm, opts);
    return {
      ...h,
      baseScore: Number.isFinite(h.score) ? h.score : null,
      ...extra
    };
  });

  scored.sort((a, b) => safeNum(b.adjustedScore) - safeNum(a.adjustedScore));

  // If cap <= 0: no per-file cap, just global max
  if (cap <= 0) return scored.slice(0, maxHits);

  const perFile = new Map();
  const out = [];

  for (const h of scored) {
    const k = h.path || "";
    const c = perFile.get(k) || 0;
    if (c >= cap) continue;

    perFile.set(k, c + 1);
    out.push(h);

    if (out.length >= maxHits) break;
  }

  return out;
}

async function retrieve(root, q, opts = {}) {
  const backendOpt = opts.backend ?? "auto";
  const contextLines = opts.contextLines ?? 1;
  const maxHits = opts.maxHits ?? 50;
  const maxSeedFiles = opts.maxSeedFiles ?? 10;
  const expand = opts.expand ?? ["imports", "dependents"];
  const maxRelated = opts.maxRelated ?? 30;

  const hitsPerFileCap = Number.isFinite(opts.hitsPerFileCap) ? opts.hitsPerFileCap : 5;
  const explainHits = !!opts.explainHits;

  const zoektBuild = !!opts.zoektBuild;
  const zoektPort = opts.zoektPort ?? 6070;

  const warnings = [];

  // health + gating
  const h = intel.health(root);
  const rerankMinResolutionPct = opts.rerankMinResolutionPct ?? 90;
  const useGraphBoost = (h.resolutionPct ?? 100) >= rerankMinResolutionPct;

  const backend = pickBackend(backendOpt, root, { zoektBuild });
  let searchRes;

  if (backend === "zoekt") {
    try {
      searchRes = await zoekt.search(root, q, {
        maxHits,
        contextLines,
        autoIndex: zoektBuild,
        port: zoektPort
      });
    } catch (e) {
      warnings.push(`zoekt failed; falling back to rg: ${e.message}`);
      searchRes = await rg.search(root, q, {
        maxHits,
        contextLines,
        mode: opts.rgMode || "literal"
      });
    }
  } else {
    searchRes = await rg.search(root, q, {
      maxHits,
      contextLines,
      mode: opts.rgMode || "literal"
    });
  }

  const hits = searchRes.hits || [];
  const byFile = groupHitsByFile(hits);
  const filePaths = [...byFile.keys()];

  // build file results
  const fileResults = [];
  for (const p of filePaths) {
    const hs = byFile.get(p) || [];
    const bestHitScore = hs.reduce(
      (m, x) => (x.score != null && x.score > m ? x.score : m),
      -Infinity
    );

    fileResults.push({
      path: p,
      id: slugForPath(p),
      type: null,
      hitCount: hs.length,
      bestHitScore: Number.isFinite(bestHitScore) ? bestHitScore : null,
      bestAdjustedHitScore: null,
      fanIn: null,
      fanOut: null,
      isEntryPoint: isEntryPoint(p),
      isHotspot: false
    });
  }

  // graph augmentation (optional)
  const g = await openGraph(root);
  let graphAvailable = false;

  if (g?.db) {
    graphAvailable = true;

    const ids = fileResults.map((f) => f.id);
    const meta = getNodeMeta(g.db, ids);
    const fanIn = getFanIn(g.db, ids);
    const fanOut = getFanOut(g.db, ids);

    for (const f of fileResults) {
      const m = meta.get(f.id);
      f.type = m?.type || "unknown";
      f.fanIn = fanIn.get(f.id) || 0;
      f.fanOut = fanOut.get(f.id) || 0;
    }

    // hotspot heuristic: top-N fanIn among the hit set
    const sortedFanIn = [...fileResults].sort((a, b) => (b.fanIn || 0) - (a.fanIn || 0));
    const cutoff = sortedFanIn[Math.min(sortedFanIn.length - 1, 4)]?.fanIn || 0;
    for (const f of fileResults) {
      f.isHotspot = (f.fanIn || 0) >= cutoff && (f.fanIn || 0) > 0;
    }
  } else {
    warnings.push("graph not available; related expansion disabled");
  }

  // Build map path -> file meta for hit reranking
  const fileByPath = new Map(fileResults.map((f) => [f.path, f]));

  // Rerank individual hits using file signals + (optionally) graph signals
  const rerankedHits = rerankAndCapHits(hits, fileByPath, {
    maxHits,
    hitsPerFileCap,
    explainHits,
    useGraphBoost
  });

  // Update fileResults with bestAdjustedHitScore (useful for debugging / UI)
  const bestAdj = new Map();
  for (const h2 of rerankedHits) {
    const prev = bestAdj.get(h2.path) ?? -Infinity;
    if (h2.adjustedScore != null && h2.adjustedScore > prev) bestAdj.set(h2.path, h2.adjustedScore);
  }
  for (const f of fileResults) {
    f.bestAdjustedHitScore = bestAdj.has(f.path) ? bestAdj.get(f.path) : null;
  }

  // Rerank files AFTER bestAdjustedHitScore is available
  rerankFiles(fileResults, { useGraphBoost });

  // pick seed files for structural expansion
  const seed = fileResults.slice(0, maxSeedFiles);

  // expand related neighbors
  const related = [];
  if (g?.db) {
    for (const s of seed) {
      const n = neighbors(g.db, s.id, { maxImports: 15, maxDependents: 15 });

      if (expand.includes("imports")) {
        for (const tid of n.imports) {
          related.push({ from: s.path, relation: "imports", id: tid });
        }
      }
      if (expand.includes("dependents")) {
        for (const sid of n.dependents) {
          related.push({ from: s.path, relation: "depended_on_by", id: sid });
        }
      }
    }

    // attach path/type/fanIn
    const relatedIds = uniq(related.map((r) => r.id));
    const meta = getNodeMeta(g.db, relatedIds);
    const fanIn = getFanIn(g.db, relatedIds);

    for (const r of related) {
      const m = meta.get(r.id);
      r.path = m?.path || null;
      r.type = m?.type || "unknown";
      r.fanIn = fanIn.get(r.id) || 0;
    }
  }

  // prune related (drop null paths, dedup by (relation,path))
  const seenRel = new Set();
  const relatedClean = [];
  for (const r of related) {
    if (!r.path) continue;
    const k = `${r.relation}:${r.path}`;
    if (seenRel.has(k)) continue;
    seenRel.add(k);
    relatedClean.push(r);
    if (relatedClean.length >= maxRelated) break;
  }

  closeGraph(g);

  // git info (optional)
  let git = null;
  try {
    const { execSync } = require("child_process");
    git = {
      branch: execSync(`git -C "${root.replace(/"/g, '\\"')}" rev-parse --abbrev-ref HEAD`, { encoding: "utf8" }).trim(),
      head: execSync(`git -C "${root.replace(/"/g, '\\"')}" rev-parse HEAD`, { encoding: "utf8" }).trim()
    };
  } catch {}

  return {
    schema: "codebase-intel.retrieve.v1",
    timestamp: new Date().toISOString(),
    repo: { root, git },
    query: {
      q,
      opts: {
        backend: backendOpt,
        contextLines,
        maxHits,
        maxSeedFiles,
        expand,
        maxRelated,
        hitsPerFileCap,
        explainHits,
        rerankMinResolutionPct
      }
    },
    providers: {
      search: { name: searchRes.provider, details: searchRes.details || null },
      graph: { available: graphAvailable }
    },
    health: h,
    results: {
      files: fileResults,
      hits: rerankedHits,
      related: relatedClean
    },
    warnings
  };
}

module.exports = { retrieve };

const chokidar = require("chokidar");
const intel = require("./intel");
const { loadRepoConfig } = require("./config");

function debounce(fn, ms) {
  let t = null;
  return (...args) => {
    clearTimeout(t);
    t = setTimeout(() => fn(...args), ms);
  };
}

async function watchOne(root, { summaryEverySecOverride = null } = {}) {
  const cfg = loadRepoConfig(root);
  const throttleMs = Math.floor(((summaryEverySecOverride ?? cfg.summaryEverySec) || 0) * 1000);

  await intel.init(root);

  const pending = new Set();
  const flush = debounce(async () => {
    const files = [...pending];
    pending.clear();
    for (const rel of files) {
      try {
        await intel.updateFile(root, rel, { summaryThrottleMs: throttleMs });
      } catch (e) {
        process.stderr.write(`[intel] ${root}: update failed ${rel}: ${e.message}\n`);
      }
    }
  }, 250);

  const watcher = chokidar.watch(cfg.globs, {
    cwd: root,
    ignoreInitial: true,
    ignored: cfg.ignore,
    persistent: true,
    awaitWriteFinish: { stabilityThreshold: 200, pollInterval: 50 }
  });

  const onChange = (rel) => { pending.add(rel); flush(); };

  watcher.on("add", onChange)
    .on("change", onChange)
    .on("unlink", (rel) => process.stderr.write(`[intel] ${root}: unlink ignored ${rel}\n`))
    .on("error", (err) => process.stderr.write(`[intel] ${root}: watcher error: ${err}\n`))
    .on("ready", () => process.stderr.write(`[intel] watching ${root}\n`));

  return watcher;
}

async function watchRoots(roots, opts = {}) {
  const watchers = [];
  for (const r of roots) watchers.push(await watchOne(r, opts));

  const shutdown = async (sig) => {
    process.stderr.write(`[intel] shutting down (${sig})\n`);
    for (const w of watchers) { try { await w.close(); } catch {} }
    process.exit(0);
  };

  process.on("SIGINT", () => shutdown("SIGINT"));
  process.on("SIGTERM", () => shutdown("SIGTERM"));

  await new Promise(() => {});
}

module.exports = { watchRoots };

function isTestPath(p) {
  if (!p) return false;
  return (
    /(^|\/)(__tests__|__test__|tests?|specs?)\//i.test(p) ||
    /\.(test|spec)\./i.test(p)
  );
}

function isVendorPath(p) {
  if (!p) return false;
  return /(^|\/)(node_modules|dist|build|coverage|\.next|out|vendor)\//i.test(p);
}

function looksLikeExportLine(line) {
  return typeof line === "string" && /^\s*export\b/.test(line);
}

function looksLikeDefinitionLine(line) {
  if (typeof line !== "string") return false;
  return (
    /^\s*(export\s+)?(default\s+)?(async\s+)?function\b/.test(line) ||
    /^\s*(export\s+)?(default\s+)?class\b/.test(line) ||
    /^\s*export\s+(const|let|var|type|interface)\b/.test(line)
  );
}

function safeNum(n, fallback = 0) {
  return Number.isFinite(n) ? n : fallback;
}

/**
 * Compute a hit score that respects provider scoring but nudges toward:
 *  - entrypoints
 *  - dependency hotspots (fan-in)
 *  - non-test, non-generated code
 *
 * Strategy: multiplicative “small percent” boosts so it scales with Zoekt score,
 * plus tiny additive nudges so rg-only results still sort meaningfully.
 */
function computeAdjustedHitScore(hit, fileMeta, opts) {
  const explain = !!opts.explainHits;

  // Base score:
  // - Zoekt often provides numeric score.
  // - rg has no score -> treat as 1 to allow multipliers to work.
  const base = Number.isFinite(hit.score) ? hit.score : 1;
  let adjusted = base;

  const signals = [];

  if (fileMeta) {
    // Boost entrypoints
    if (fileMeta.isEntryPoint) {
      adjusted += base * 0.10; // +10%
      if (explain) signals.push("entrypoint:+10%");
    }

    // Boost hotspots
    if (fileMeta.isHotspot) {
      adjusted += base * 0.15; // +15%
      if (explain) signals.push("hotspot:+15%");
    }

    // Boost by fan-in (log scaled, capped)
    const fanIn = safeNum(fileMeta.fanIn, 0);
    if (fanIn > 0) {
      const frac = Math.min(0.15, Math.log1p(fanIn) * 0.02); // up to +15%
      adjusted += base * frac;
      if (explain) signals.push(`fanin:${fanIn}:+${Math.round(frac * 100)}%`);
    }

    // Penalize tests
    if (isTestPath(fileMeta.path)) {
      adjusted -= base * 0.25; // -25%
      if (explain) signals.push("test:-25%");
    }

    // Penalize vendor/build outputs (should rarely be top results)
    if (isVendorPath(fileMeta.path)) {
      adjusted -= base * 0.50; // -50%
      if (explain) signals.push("vendor:-50%");
    }
  }

  // Small line-content nudges (helps when many matches)
  if (looksLikeExportLine(hit.line)) {
    adjusted += base * 0.05;
    if (explain) signals.push("exportline:+5%");
  }
  if (looksLikeDefinitionLine(hit.line)) {
    adjusted += base * 0.03;
    if (explain) signals.push("defline:+3%");
  }

  // Tiny additive nudges so rg results don’t all tie perfectly
  if (!Number.isFinite(hit.score)) {
    if (fileMeta?.isEntryPoint) adjusted += 0.20;
    if (fileMeta?.isHotspot) adjusted += 0.25;
    adjusted += Math.min(0.25, Math.log1p(safeNum(fileMeta?.fanIn, 0)) * 0.05);
  }

  // Clamp (avoid negative scores)
  adjusted = Math.max(0, adjusted);

  return explain ? { adjustedScore: adjusted, signals } : { adjustedScore: adjusted };
}

/**
 * Sort hits by adjustedScore and cap hits per file for diversity.
 */
function rerankAndCapHits(hits, fileByPath, opts) {
  const maxHits = safeNum(opts.maxHits, hits.length);
  const cap = Number.isFinite(opts.hitsPerFileCap) ? opts.hitsPerFileCap : 5;

  const scored = hits.map(h => {
    const fm = fileByPath.get(h.path);
    const extra = computeAdjustedHitScore(h, fm, opts);
    return {
      ...h,
      baseScore: Number.isFinite(h.score) ? h.score : null,
      ...extra
    };
  });

  scored.sort((a, b) => safeNum(b.adjustedScore) - safeNum(a.adjustedScore));

  if (cap <= 0) return scored.slice(0, maxHits);

  const perFile = new Map();
  const out = [];

  for (const h of scored) {
    const k = h.path || "";
    const c = perFile.get(k) || 0;
    if (c >= cap) continue;

    perFile.set(k, c + 1);
    out.push(h);

    if (out.length >= maxHits) break;
  }

  return out;
}

5) Tuning (quick guidance)

If you find tests still show up too much:
	•	increase test:-25% to -35%

If Zoekt scores swamp the structural boosts:
	•	increase hotspot boost from +15% to +25%
	•	increase entrypoint boost from +10% to +15%

Keep boosts percentage-based; it’s the simplest way to stay compatible across Zoekt score scales.
